Table users {
  id bigint [pk, increment]
  email varchar [not null, unique]
  password varchar [not null]
  nickname varchar [not null]
  profile_image varchar
  role user_role [not null]
  created_at timestamp [not null]
}
// =====================================
// COURSE DOMAIN
// =====================================
Table categories {
  id bigint [pk, increment]
  parent_category_id bigint
  name varchar [not null]
  depth int [not null]
}
Table courses {
  id bigint [pk, increment]
  instructor_id bigint [not null]
  category_id bigint
  title varchar [not null]
  description text
  thumbnail varchar
  price int [not null]
  status course_status [not null]
  view_count int [not null, default: 0]
  created_at timestamp [not null]
  updated_at timestamp [not null]
  deleted_at timestamp
}
Table course_status_histories {
  id bigint [pk, increment]
  course_id bigint [not null]
  from_status course_status
  to_status course_status
  reason text
  changed_by bigint [not null]
  created_at datetime [not null]
}
Table chapters {
  id bigint [pk, increment]
  course_id bigint [not null]
  title varchar [not null]
  order_no int [not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
  deleted_at timestamp
}
Table lectures {
  id bigint [pk, increment]
  chapter_id bigint [not null]
  m3u8_path varchar [not null]
  title varchar [not null]
  summary varchar [null]
  duration_seconds int [not null]
  order_no int [not null]
  is_free_preview boolean [not null, default: false]
  created_at timestamp [not null]
  updated_at timestamp [not null]
  deleted_at timestamp
}
Table lecture_modification_requests {
  id bigint [pk, increment]
  course_id bigint [not null]       // 어떤 강좌의
  lecture_id bigint [not null]      // 어떤 강의 영상이 바뀌는지
  instructor_id bigint [not null]   // 신청한 강사 식별자
  
  description text [not null]       // ★ 판매자가 직접 작성하는 "영상 수정 사유/내역"
  
  before_m3u8_path varchar [not null] // 변경 전 기존 구버전 영상 UUID 경로
  after_m3u8_path varchar [not null]  // FFmpeg 변환이 완료된 신버전 영상 UUID 경로
  
  status varchar [not null]         // 신청 상태 (PENDING, APPROVED, REJECTED)
  rejected_reason text              // 관리자가 반려할 경우 적는 반려 사유
  
  managed_by bigint                 // 검토한 관리자 식별자
  created_at timestamp [not null]   // 신청 일시
  updated_at timestamp [not null]   // 처리 일시
}
